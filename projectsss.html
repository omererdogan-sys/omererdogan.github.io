// Example: formatted local time
func localTime(for timeZoneId: String) -> String {
    let tz = TimeZone(identifier: timeZoneId) ?? .current
    let formatter = DateFormatter()
    formatter.timeZone = tz
    formatter.dateStyle = .none
    formatter.timeStyle = .medium
    return formatter.string(from: Date())
}

Cryptography

Caesar cipher, encoding/decoding basics.

func caesar(_ text: String, shift: Int) -> String {
    let a = Array(text.unicodeScalars)
    return String(a.map { scalar in
        let v = Int(scalar.value)
        let base = (65...90).contains(v) ? 65 : (97...122).contains(v) ? 97 : v
        guard base != v else { return Character(scalar) }
        let shifted = (v - base + shift) % 26 + base
        return Character(UnicodeScalar(shifted)!)
    })
}

Decimal‑Binary Converter

Convert integers between decimal and binary.

func toBinary(_ n: Int) -> String {
    var n = n; var bits: [Int] = []
    repeat { bits.append(n % 2); n /= 2 } while n > 0
    return bits.reversed().map(String.init).joined()
}
